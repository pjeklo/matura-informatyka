ŚCIĄGA Z PYTHONA

1. PODSTAWOWE METODY ODCZYTU PLIKÓW

Odczyt całego pliku

# Cały plik jako string
with open('plik.txt', 'r') as f:
    zawartosc = f.read()

# Cały plik jako lista linii
with open('plik.txt', 'r') as f:
    linie = f.readlines()

Odczyt linia po linii

# Iteracja po liniach
with open('plik.txt', 'r') as f:
    for linia in f:
        print(linia.strip())  # usuwa białe znaki na początku/końcu

2. ODCZYT PLIKÓW CSV

Podstawowy odczyt CSV

# Odczyt bez użycia bibliotek
with open('dane.csv', 'r') as f:
    for linia in f:
        wartosci = linia.strip().split(',')
        print(wartosci)

# Odczyt z podziałem na kolumny
with open('dane.csv', 'r') as f:
    naglowki = f.readline().strip().split(',')
    dane = [linia.strip().split(',') for linia in f]

Zaawansowany odczyt CSV z biblioteką

import csv

# Odczyt całego pliku
with open('dane.csv', 'r') as f:
    czytnik = csv.reader(f)
    for wiersz in czytnik:
        print(wiersz)

# Odczyt do słownika
with open('dane.csv', 'r') as f:
    czytnik = csv.DictReader(f)
    for wiersz in czytnik:
        print(wiersz)  # wiersz jako słownik

3. ODCZYT PLIKÓW TEKSTOWYCH Z KODOWANIEM

# Odczyt pliku z określonym kodowaniem
with open('plik.txt', 'r', encoding='utf-8') as f:
    zawartosc = f.read()

# Obsługa różnych kodowań
try:
    with open('plik.txt', 'r', encoding='utf-8') as f:
        zawartosc = f.read()
except UnicodeDecodeError:
    with open('plik.txt', 'r', encoding='latin-1') as f:
        zawartosc = f.read()

4. ZAAWANSOWANE OPERACJE NA PLIKACH

Filtrowanie podczas odczytu

# Odczyt z filtrowaniem
with open('dane.txt', 'r') as f:
    dane_filtrowane = [
        linia.strip() for linia in f 
        if not linia.startswith('#')  # pomijanie komentarzy
    ]

Wczytywanie plików dużych rozmiarów

# Bezpieczny odczyt dużego pliku
def czytaj_duzy_plik(nazwa_pliku, rozmiar_bufora=1024):
    with open(nazwa_pliku, 'r') as f:
        while True:
            fragment = f.read(rozmiar_bufora)
            if not fragment:
                break
            # Przetwarzaj fragment
            print(fragment)

5. ZAPIS DO PLIKU

# Podstawowy zapis
with open('wynik.txt', 'w') as f:
    f.write('Tekst do zapisu\n')

# Dopisywanie do pliku
with open('wynik.txt', 'a') as f:
    f.write('Kolejna linia\n')

# Zapis listy do pliku
dane = ['linia1', 'linia2', 'linia3']
with open('wynik.txt', 'w') as f:
    for linia in dane:
        f.write(f'{linia}\n')

6. OBSŁUGA BŁĘDÓW PRZY PRACY Z PLIKAMI

# Kompleksowa obsługa błędów
try:
    with open('plik.txt', 'r') as f:
        zawartosc = f.read()
except FileNotFoundError:
    print('Plik nie istnieje')
except PermissionError:
    print('Brak uprawnień do odczytu')
except IOError as e:
    print(f'Błąd wejścia/wyjścia: {e}')

8. PRAKTYCZNE PRZYKŁADY ZASTOSOWAŃ

Analiza logu

# Zliczanie wystąpień
def analiza_logu(nazwa_pliku, szukana_fraza):
    licznik = 0
    with open(nazwa_pliku, 'r') as f:
        for linia in f:
            if szukana_fraza in linia:
                licznik += 1
    return licznik

Przetwarzanie pliku konfiguracyjnego

# Prosta konfiguracja z pliku
def wczytaj_konfiguracje(nazwa_pliku):
    konfiguracja = {}
    with open(nazwa_pliku, 'r') as f:
        for linia in f:
            if '=' in linia:
                klucz, wartosc = linia.strip().split('=')
                konfiguracja[klucz.strip()] = wartosc.strip()
    return konfiguracja
